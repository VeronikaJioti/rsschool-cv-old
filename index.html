DECLARE @bpid INT
DECLARE Curs CURSOR FOR
 SELECT BPId FROM dbo.BusinessProcesses  WHERE BPTypeCode='REVENUE_CLIENT_CLOSE_OUT' AND ValueDate='20130130' AND BPStatusId=9
OPEN Curs
FETCH NEXT FROM Curs INTO @bpid
WHILE @@FETCH_STATUS=0
 BEGIN
 UPDATE dbo.BusinessProcesses
 SET BPStatusId=10
 WHERE BPId=@bpid
FETCH NEXT FROM Curs INTO @bpid
END
CLOSE Curs
DEALLOCATE Curs
